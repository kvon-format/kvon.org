# inlined arrays
array: [1 true 'str' [true false]]

# multi-dimensional arrays
multi-dim-array:--
	# primitives and inlined arrays can be
	# placed on one line
	- 'str 1' 'str 2' [0 1 [true false]]
	# objects and multi line strings have to be
	# placed on separate lines
	- key: 'value'
	- |
		line 1
		line 2
		line 3
